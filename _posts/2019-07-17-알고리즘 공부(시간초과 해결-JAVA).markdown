Stack 관련된 문제를 풀다가 어려움에 직면하였다.<br>
문제에 주어진 예제는 모두 해결하는 알고리즘을 짰으나 문제에서 요구하는 제한시간을 통과하지 못하였다.
아직도 해결하지 못해서 내 정신상태가 이상해지는 것 같다.

<a href="https://www.acmicpc.net/problem/1874">[백준 1874번]</a><br>

<pre>
import java.io.*;

public class Main {
	public static void main(String[] args) throws IOException {

		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
		StringBuilder sb = new StringBuilder();
		int k = 1;

		int number = Integer.parseInt(br.readLine());
		int[] inputNumber = new int[number];
		stack s = new stack(number);

		int n = 1;
		for (int i = 0; i < number;) {
			int input = Integer.parseInt(br.readLine());

			if (n <= input) {
				while (n <= input) {
					s.push(n);
					sb.append("+\n");
					n++;
				}
				if (s.top() == input) {
					s.pop();
					sb.append("-\n");
					i++;
				}
			} else if (n > input) {
				if (s.top() == input) {
					s.pop();
					sb.append("-\n");
					i++;
				} else {
					k = 0;
					bw.write("NO");
					bw.flush();
					bw.close();
					break;
				}
			}
		}

		if (k == 1) {
			bw.write(sb.toString());
			bw.flush();
			bw.close();
		}
	}
}

class stack {
	int[] stack;
	int size = 0;

	public stack(int n) {
		stack = new int[n];
	}

	void push(int x) {
		stack[size++] = x;
	}

	int pop() {
		return stack[--size];
	}

	int top() {
		return stack[size - 1];
	}

	int size() {
		return size;
	}

	int empty() {
		if (size == 0)
			return 1;
		else
			return 0;
	}
}

### JAVA - BufferedReader, BufferedWriter 를 통한 입출력

> <pre>
> import java.io.*;
> import java.util.StringTokenizer;
> 
> public class Main {
> 	public static void main(String[] args) throws IOException {
> 		BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));
> 		BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
> 
> 		int T = Integer.parseInt(bf.readLine());// 테스트케이스의 개수 T(최대 1000,000)
> 		for (int i = 0; i < T; i++) {
> 			String s = bf.readLine();
> 			StringTokenizer st = new StringTokenizer(s);
> 			int A = Integer.parseInt(st.nextToken());
> 			int B = Integer.parseInt(st.nextToken());
> 			bw.write(String.valueOf(A+B+"\n"));
> 		}
> 		bw.flush();
> 		bw.close();
> 	}
> }
> </pre>
